@startuml
title Отображение топа песен
autonumber

actor Пользователь as User
participant "Frontend" as Frontend
participant "Backend" as Backend
database "DB" as DataBase

activate User
User -> Frontend: Нажатие на кнопку "тренды"

activate Frontend
Frontend -> Backend: Запрос данных пользователя

activate Backend
Backend -> DataBase: Запрос данных пользователя

activate DataBase
DataBase --> Backend: Ответ с топом песен
deactivate DataBase

Backend --> Frontend: Ответ с топом песен
deactivate Backend

Frontend --> User: Отобразить перечень топа песен
deactivate Frontend
deactivate User
@enduml
