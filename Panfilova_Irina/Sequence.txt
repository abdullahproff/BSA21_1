@startuml

skin rose

actor User
participant frontend
participant backend
database DB

activate User
User -> frontend: 1. нажать кнопку "добавить тег"
activate frontend

frontend -> backend: 2. HTTP-запрос на добавление тега
activate backend

backend -> backend: 3. валидация запроса
backend -> DB: 4. SQL - запрос на получение данных

activate DB
DB -> backend: 5. SQL - ответ со списком доступных тегов
deactivate DB

backend -> frontend: 6. возвращает json со списком тегов
deactivate backend

frontend -> User: 7. отображает список доступных тегов
deactivate frontend

deactivate user

@enduml
